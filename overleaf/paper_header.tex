\newtheorem{hypothesis}{Hypothesis}

\makeatletter
\newcounter{subhyp}
\let\savedc@hypothesis\c@hypothesis
\newenvironment{subhyp}
{%
	\setcounter{subhyp}{0}%
	\stepcounter{hypothesis}%
	\edef\saved@hypothesis{\thehypothesis}% Save the current value of hyp
	\let\c@hypothesis\c@subhyp     % Now hyp is subhyp
	\renewcommand{\thehypothesis}{\saved@hypothesis\alph{hypothesis}}%
}
{}
\newcommand{\normhyp}{%
	\let\c@hypothesis\savedc@hypothesis % revert to the old one
	\renewcommand\thehypothesis{\arabic{hypothesis}}%
}
\makeatother

\newcommand*{\addFileDependency}[1]{% argument=file name and extension
  \typeout{(#1)}
  \@addtofilelist{#1}
  \IfFileExists{#1}{}{\typeout{No file #1.}}
}
\makeatother

\newcommand*{\myexternaldocument}[1]{%
    \externaldocument{#1}%
    \addFileDependency{#1.tex}%
    \addFileDependency{#1.aux}%
}

    \begin{titlingpage}

\newcommand{\appendixTOC}{
	\thispagestyle{empty}
	\onehalfspacing

	\pagenumbering{arabic}
	\setcounter{table}{0}
	\setcounter{figure}{0}
	\setcounter{equation}{0}
	\renewcommand*{\thepage}{A\arabic{page}}
	\renewcommand{\thetable}{A\arabic{table}}
	\renewcommand{\thefigure}{A\arabic{figure}}
	\renewcommand{\theequation}{A\arabic{equation}}

	\addtocontents{toc}{\protect\setcounter{tocdepth}{3}}

	{\hypersetup{linkcolor=black}
		\def\contentsname{\empty}
		\tableofcontents
	}

	\clearpage
	\setcounter{page}{1}
}
